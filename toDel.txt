// create the smooth scroller FIRST!

 wrapper: "#wrapper",
 content: "#content",
 smooth: 2,
 speed: 3,
 effects: true
});

smoother.effects(".hero__image-cont", {
 speed: () => gsap.utils.random(0.55, 0.85, 0.05)
});

 yPercent: 300,

 },

});

 scale: 1.5,
 xPercent: 20,
 scrollTrigger: {
  trigger: ".hero",
  start: "top top",
  end: "+=3000px",

 }
});
